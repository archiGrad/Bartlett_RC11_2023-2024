% Define geometry settings for even (left) pages

}

    right=1cm,

}




%pagenumbers
% Define page style
\fancypagestyle{mypagestyle}{
    \fancyhf{} % Clear header and footer
    \fancyfoot[LE]{\thepage\hspace{1em}\leftmark} % Right side on odd pages, left side on even pages
    \fancyfoot[RO]{\rightmark\hspace{1em}\thepage} % Left side on even pages, right side on odd pages
    \renewcommand{\footrulewidth}{0pt} % Remove footer rule
    \renewcommand{\headrulewidth}{0pt} % Remove header rule
}
